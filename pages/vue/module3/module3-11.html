3.11. Добавление маршрутизатора в проект

Далее нам необходимо добавить маршрутизатор. Добавим файл routes.js с кодом:

export default [
 {
   path: '/',
   name: 'IndexHome',
   component: () => import('../views/Index.vue'),
   meta: { layout: 'AppLayoutMain' },
   children: [
     {
       path: '/:id',
       name: 'TaskView',
       component: () => import('../views/TaskView.vue'),
       meta: { layout: 'AppLayoutMain' }
     }
   ]
 },
 {
   path: '/tasks/create',
   name: 'TaskCreate',
   component: () => import('../views/TaskCreate.vue'),
   meta: { layout: 'AppLayoutDefault' }
 },
 {
   path: '/tasks/edit/:id',
   name: 'TaskEdit',
   component: () => import('../views/TaskEdit.vue'),
   meta: { layout: 'AppLayoutDefault' }
 }
];

Мы добавили четыре маршрута:

  • IndexHome — наша домашняя страница, которая отображает список задач.
  • TaskView — страница, которая будет отображать отдельную задачу.
  • TaskCreate — страница, на которой администратор сможет создавать новые задачи.
  • TaskEdit — станица для редактирования существующих задач.

Мы пока не создали страницы TaskView, TaskCreate, TaskEdit, но сделаем это в ближайшее время.

Далее нам необходимо создать экземпляр маршрутизатора vue-router и подключить его в main.js. Добавим следующий код в файл router/index.js:

import Vue from 'vue';
import Router from 'vue-router';
import routes from '@/router/routes';

Vue.use(Router);

export default new Router({
 mode: 'history',
 base: process.env.BASE_URL,
 routes
});

Далее маршрутизатор необходимо подключить в main.js:

import Vue from 'vue';
import App from '@/App.vue';
import '@/plugins/ui';
import router from './router';

Vue.config.productionTip = false;

new Vue({
 router,
 render: h => h(App)
}).$mount('#app');

Также необходимо добавить тег <router-view> в App.vue:

<template>
  <div id="app">
    <AppLayout
      :tasks="filteredTasks"
      :filters="filters"
      @updateTasks="updateTasks"
      @applyFilters="applyFilters"
    >
+      <router-view
+        :tasks="routeProps.tasks"
+        :filters="routeProps.filters"
+        @updateTasks="updateTasks"
+        @applyFilters="applyFilters"
+        @addTask="addTask"
+        @editTask="editTask"
+        @deleteTask="deleteTask"
+      />
    </AppLayout>
  </div>
</template>
<script>
- import AppLayout from '@/layouts/AppLayout';
+ import users from '@/static/users.json';
import tasks from '@/static/tasks.json';
import { normalizeTask } from '@/common/helpers';

export default {
  name: 'App',
-  components: {
-    AppLayout
-  },
  data() {
    return {
      tasks: tasks.map(task => normalizeTask(task)),
      filters: {
        search: '',
        users: [],
        statuses: []
      }
    };
  },
  computed: {
+    // Временное решение для передачи данных до подключения vuex
+    routeProps() {
+      const routes = {
+        IndexHome: { tasks: this.filteredTasks, filters: this.filters },
+        TaskView: { tasks: this.filteredTasks, filters: this.filters },
+        TaskEdit: { tasks: this.filteredTasks, filters: null },
+        TaskCreate: { tasks: null, filters: null }
+      };
+      return routes[this.$route.name] || {};
+    },
    filteredTasks() {
      const filtersAreEmpty = Object.values(this.filters)
        .every(value => !value.length);
      if (filtersAreEmpty) {
        // Note: to avoid tasks filtering if the filter are not set
        return this.tasks;
      }

      const searchFilter = task => task.title
        .toLowerCase()
        .includes(this.filters.search.toLowerCase().trim());

      const usersFilter = task => this.filters.users
        .some(userId => userId === task.userId);

      const statusesFilter = task => this.filters.statuses
        .some(el => el === task.status || el === task.timeStatus);

      return this.tasks.filter(task => {
        let result = {
          search: searchFilter,
          users: usersFilter,
          statuses: statusesFilter
        };
        return Object.entries(result)
          .every(([key, callback]) =>
            !this.filters[key].length || callback(task));
      });
    }
  },
  methods: {
    updateTasks(tasksToUpdate) {
      tasksToUpdate.forEach(task => {
        const index = this.tasks.findIndex(({ id }) => id === task.id);
        if (~index) {
          this.tasks.splice(index, 1, task);
        }
      });
    },
    applyFilters({ item, entity }) {
      if (!Array.isArray(this.filters[entity])) {
        this.filters[entity] = item;
      } else {
        const resultValues = [...this.filters[entity]];
        const itemIndex = resultValues.findIndex(el => el === item);
        ~itemIndex
          ? resultValues.splice(itemIndex, 1)
          : resultValues.push(item);
        this.filters[entity] = resultValues;
      }
    },
+    addTask(task) {
+      const newTask = normalizeTask(task);
+      newTask.id = this.tasks.length + 1;
+      newTask.sortOrder = this.tasks.filter(task => !task.columnId).length;
+      if (newTask.userId) {
+        newTask.user = { ...this.getTaskUser(newTask.userId) };
+      }
+      this.tasks = [...this.tasks, newTask];
+    },
+    editTask(task) {
+      const index = this.tasks.findIndex(({ id }) => task.id === id);
+      if (~index) {
+        const newTask = normalizeTask(task);
+        if (newTask.userId) {
+          newTask.user = { ...this.getTaskUser(newTask.userId) };
+        }
+        this.$set(this.tasks, index, newTask);
+      }
+    },
+    deleteTask(id) {
+      this.tasks = this.tasks.filter(task => task.id !== id);
+    },
+    getTaskUser(userId) {
+      return users.find(user => user.id === userId);
+    }
  }
};
</script>

 name: 'TaskCardView',
 components: {
   TaskCardViewTicksList,
   TaskCardTags,
   TaskCardViewComments
 },
 mixins: [taskCardName],

   tasks: {
     type: Array,
     required: true
   }
 },
 data: () => ({
   task: null
 }),
 computed: {
   user() {
     return users[0];
   },
   dueDate() {
     if (!this.task) {
       return false;
     }
     const { dueDate } = this.task;
     return dueDate ? getReadableDate(dueDate) : false;

     return getTimeAgo(this.task?.dueDate);

 created() {
   const id = this.$route.params.id;

     this.$router.push('/').catch(() => {});
   }
 },

   this.$refs.dialog?.focus();
 },
 methods: {
   addCommentToList(comment) {
     this.task.comments
       ? this.task.comments.push(comment)
       : this.$set(this.task, 'comments', [comment]);
   },
   closeDialog() {
     this.$router.push('/');
   }
 }
};
</script>

<script>
import { validator } from '@/common/mixins';
import users from '@/static/users.json';

export default {
 name: 'TaskCardViewComments',
 mixins: [validator],
 props: {
   taskId: {
     type: Number,
     required: true
   },
   comments: {
     type: Array,
     default: () => []
   }
 },
 data: () => ({
   newComment: '',
   validations: {
     newComment: {
       error: '',
       rules: ['isNotEmpty']
     }
   }
 }),
 computed: {
   user() {
     return users[0];
   }
 },
 watch: {
   newComment() {
     this.$clearValidationErrors();
   }
 },
 methods: {
   submit() {
     if (!this.$validateFields(
       { newComment: this.newComment }, this.validations)
     ) {
       return;
     }
     const { id, name, avatar } = this.user;
     const comment = {
       text: this.newComment,
       taskId: this.taskId,
       userId: id
     };
     this.newComment = '';
     this.$emit('new-comment', {
       ...comment,
       user: { id, name, avatar }
     });
   }
 }
};
</script>

<script>
export default {
 name: 'TaskCardViewTicksList',
 props: {
   ticks: {
     type: [Array],
     default: () => []
   },
   disabled: {
     type: Boolean,
     default: false
   }
 },
 methods: {
   updateTick(tick, property, value) {
     const tickCopy = Object.assign({}, tick);
     tickCopy[property] = value;
     this.$emit('updateTick', tickCopy);
   }
 }
};
</script>

Мой наставник

Для полноценного прохождения курса вам нужно выбрать наставника.

После выбора вы сможете всегда держать под рукой его контакты (почту и ссылки на профиль в соцсетях, которые он укажет для связи со своими студентами), а также расписание консультаций.

Вы в любой момент можете поменять наставника, если вам кажется, что с текущим наладить работу не получается.

Мои проекты

Для удобства работы с проектами, мы собрали их в этом блоке.

Основной личный проект, который вы будете готовить к финальной защите, находится в начале списка и рядом с ним стоит иконка . Если на курсе используется учебный проект, то он будет следующим и с иконкой  . Далее будут идти все остальные проекты, если для них были созданы репозитории.

Вы можете выбрать, какой именно проект будет основным, в любой момент прохождения курса, но мы советуем определиться с выбором как можно скорее.

Оцените наставника

Мы всегда стремимся улучшать качество услуг. И нам важно знать, как идёт ваша работа с наставником.

Уделите пару секунд вашего времени и дайте нам знать, как обстоят дела.

Также вы всегда можете задать свои вопросы напрямую куратору.

Ваши первые задачи

В этом списке мы собрали задачи, с которых начинается обучение на курсе.

Выполняйте задачи по порядку, а система будет автоматически отмечать галочками ваш прогресс. Когда весь список будет выполнен, этот блок закроется и больше не будет вам надоедать.

Стандартный тариф

Студенты стандартного тарифа работают с наставником в течение первых пяти недель обучения. Оставшееся время консультаций, можно потратить на подготовку проекта к защите.

Запись на консультацию

Голосовая консультация — это живое общение с наставником, который ответит на интересующие вас вопросы.

Текстовая консультация — это общение с наставником в чате. Чат доступен 1 день с момента начала консультации.

Кодревью — это проверка пулреквеста в репозитории вашего проекта. Наставник оставит комментарии к вашему коду.

От выбранного раздела зависит, кто из наставников сможет провести консультацию.

Вы можете отправить заявку на консультацию конкретному наставнику, или её возьмёт первый, освободившийся из живой очереди.
