3.12. Создание компонентов для создания/редактирования задач

Давайте начнём с добавления нового представления для создания задач. Добавьте файл /views/TaskCreate.vue:

<template>
 <TaskCardCreator
   @addTask="$emit('addTask', $event)"
 />
</template>

<script>
import TaskCardCreator from '@/modules/tasks/components/TaskCardCreator';

export default {
 name: 'TaskCreate',
 components: {
   TaskCardCreator
 }
};
</script>

И компонент для редактирования задач в файле /views/TaskEdit.vue:

<template>
 <TaskCardCreator
   v-if="task"
   :task-to-edit="task"
   @editTask="$emit('editTask', $event)"
   @deleteTask="$emit('deleteTask', $event)"
 />
</template>

<script>
import TaskCardCreator from '@/modules/tasks/components/TaskCardCreator';
import { createNewDate } from '@/common/helpers';

export default {
 name: 'TaskEdit',
 components: { TaskCardCreator },
 props: {
   tasks: {
     type: Array,
     default: () => []
   }
 },
 data() {
   return {
     task: null
   };
 },
 created () {
   const id = this.$route.params.id;
   const task = this.tasks.find(task => +task.id === +id);
   if (task) {
     task.dueDate = task.dueDate ? new Date(task.dueDate) : createNewDate();
     task.ticks = task.ticks || [];
     this.task = task;
   } else {
     this.$router.push('/');
   }
 }
};
</script>

В компоненте редактирования задачи мы используем хук жизненного цикла created, в котором мы получаем задачу с сервера, и добавляем его в компонент TaksCardCreator. На данном этапе у нас ещё нет подключения к серверу (мы добавим его дальше), поэтому мы ищем редактируемую задачу в общем списке задач. После добавления сервера мы будем отправлять запрос на сервер для получения экземпляра задачи.

В обоих компонентах мы добавили внутренний компонент TaskCardCreator.vue. Давайте создадим и его:

<script>
import TaskCardCreatorUserSelector
 from '@/modules/tasks/components/TaskCardCreatorUserSelector';
import TaskCardCreatorDueDateSelector
 from '@/modules/tasks/components/TaskCardCreatorDueDateSelector';
import TaskCardViewTicksList
 from '@/modules/tasks/components/TaskCardViewTicksList';
import TaskCardTags from '@/modules/tasks/components/TaskCardTags';
import { statuses } from '@/common/constants';
import { createUUIDv4, getTimeAgo } from '@/common/helpers';
import validator from '@/common/mixins/validator';
import taskCardName from '@/common/mixins/taskCardName';
import taskStatuses from '@/common/enums/taskStatuses';
import { cloneDeep } from 'lodash';
import { createNewDate } from '@/common/helpers';

const createNewTask = () => ({
 userId: null,
 columnId: null,
 statusId: null,
 title: 'Название задачи',
 description: '',
 sortOrder: 0,
 dueDate: createNewDate(),
 url: '',
 urlDescription: '',
 ticks: [],
 tags: ''
});

const createNewTick = () => ({
 uuid: createUUIDv4(),
 taskId: null,
 text: '',
 done: false
});

export default {
 name: 'TaskCardCreator',
 components: {
   TaskCardCreatorUserSelector,
   TaskCardCreatorDueDateSelector,
   TaskCardViewTicksList,
   TaskCardTags
 },
 mixins: [validator, taskCardName],
 props: {
   taskToEdit: {
     type: Object,
     default: null
   }
 },
 data() {
   return {
     task: createNewTask(),
     validations: {
       title: {
         error: '',
         rules: ['required']
       },
       url: {
         error: '',
         rules: ['url']
       }
     },
     isFormValid: true,
     statusList: statuses.slice(0, 3),
     taskStatuses
   };
 },
 computed: {
   timeAgo() {
     return getTimeAgo(this.task?.dueDate);
   }
 },
 watch: {
   task: {
     deep: true,
     handler() {
       this.isFormValid = true;
     }
   }
 },
 created() {
   if (this.taskToEdit) {
     this.task = cloneDeep(this.taskToEdit);
   }
 },
 mounted() {
   this.$refs.dialog.focus();
 },
 methods: {
   async submit() {
     if (!this.$validateFields(this.task, this.validations)) {
       this.isFormValid = false;
       return;
     }
     this.$emit(`${this.taskToEdit ? 'editTask' : 'addTask'}`, this.task);
     this.closeDialog();
   },
   createTick() {
     this.task.ticks = [...this.task.ticks, createNewTick()];
   },
   async updateTick(tick) {
     const index = this.task.ticks
       .findIndex(({ uuid }) => tick.uuid === uuid);
     if (~index) {
       this.task.ticks.splice(index, 1, tick);
     }
   },
   async removeTick({ uuid }) {
     this.task.ticks = this.task.ticks.filter(tick => tick.uuid !== uuid);
   },
   async removeTask() {
     this.$emit('deleteTask', this.task.id);
     this.$router.push('/').catch(() => {});
   },
   setStatus(status) {
     const [key] = Object.entries(taskStatuses)
       .find(([_, value]) => value === status);
     if (!this.task.statusId || this.task.statusId !== +key) {
       this.$set(this.task, 'statusId', +key);
     } else {
       this.$set(this.task, 'statusId', null);
     }
   },
   closeDialog() {
     this.$router.push('/');
   }
 }
};
</script>

Рассмотрим подробнее данный компонент.

В шаблоне (template) мы используем некоторые вспомогательные компоненты, которые мы добавим позже.

В скрипте мы добавили две функции-фабрики createNewTask, createNewTick, которые будут создавать нам новые задачи и элементы списка.

В хуке жизненного цикла created мы проверяем, передаётся ли в компонент входной параметр (props) taskToEdit. Так компонент понимает, нужно ли создавать новую задачу или следует редактировать существующую.

В хуке жизненного цикла mounted мы передаём фокус на корневой элемент компонента.

В методах компонента мы видим методы для манипуляций с элементами списка задач, на которых мы не будем подробно останавливаться, но рассмотрим метод submit, который отвечает за создание или редактирование задачи.

Сначала нам необходимо добавить новый миксин для валидации полей. Добавим новый файл /common/mixins/validator.js:

import { emailRegex, urlRegex } from '@/common/constants';

const rules = {
 isNotEmpty: {
   rule: value => !!value?.trim(),
   message: 'Поле не заполнено'
 },
 required: {
   rule: value => !!value?.trim(),
   message: 'Поле обязательно для заполнения'
 },
 email: {
   rule: value => !!value
     ? emailRegex.test(String(value).toLowerCase())
     : true,
   message: 'Электронная почта имеет неверный формат'
 },
 url: {
   rule: value => !!value ? urlRegex.test(value) : true,
   message: 'Ссылка имеет неверный формат'
 }
};

const validator = (value, appliedRules) => {
 let error = '';
 appliedRules.forEach(appliedRule => {
   if (!rules[appliedRule]) {
     return;
   }
   const { rule, message } = rules[appliedRule];
   if (!rule(value)) {
     error = message;
   }
 });
 return error;
};

export default {
 methods: {
   $validateFields(fields, validations) {
     let isValid = true;
     Object.keys(validations).forEach(key => {
       validations[key].error = validator(
         fields[key],
         validations[key].rules
       );
       if (validations[key].error) {
         isValid = false;
       }
     });
     return isValid;
   },
   $clearValidationErrors() {
     if (!this.validations) {
       return;
     }
     Object.keys(this.validations).forEach(key => {
       this.$set(this.validations[key], 'error', '');
     });
   }
 }
};

Мы временно отключили линтер, так как он не позволяет нам сохранить длинные регулярные выражения.

Новый метод $validateFields отвечает за валидацию полей и принимает два аргумента: список полей для валидации и правила валидации. Мы также будем напрямую изменять свойство компонента validations.

Вернёмся к методу submit. После валидации полей метод эмитит событие addTask или editTask в зависимости от входного параметра (props) taskToEdit.

Мой наставник

Для полноценного прохождения курса вам нужно выбрать наставника.

После выбора вы сможете всегда держать под рукой его контакты (почту и ссылки на профиль в соцсетях, которые он укажет для связи со своими студентами), а также расписание консультаций.

Вы в любой момент можете поменять наставника, если вам кажется, что с текущим наладить работу не получается.

Мои проекты

Для удобства работы с проектами, мы собрали их в этом блоке.

Основной личный проект, который вы будете готовить к финальной защите, находится в начале списка и рядом с ним стоит иконка . Если на курсе используется учебный проект, то он будет следующим и с иконкой  . Далее будут идти все остальные проекты, если для них были созданы репозитории.

Вы можете выбрать, какой именно проект будет основным, в любой момент прохождения курса, но мы советуем определиться с выбором как можно скорее.

Оцените наставника

Мы всегда стремимся улучшать качество услуг. И нам важно знать, как идёт ваша работа с наставником.

Уделите пару секунд вашего времени и дайте нам знать, как обстоят дела.

Также вы всегда можете задать свои вопросы напрямую куратору.

Ваши первые задачи

В этом списке мы собрали задачи, с которых начинается обучение на курсе.

Выполняйте задачи по порядку, а система будет автоматически отмечать галочками ваш прогресс. Когда весь список будет выполнен, этот блок закроется и больше не будет вам надоедать.

Стандартный тариф

Студенты стандартного тарифа работают с наставником в течение первых пяти недель обучения. Оставшееся время консультаций, можно потратить на подготовку проекта к защите.

Запись на консультацию

Голосовая консультация — это живое общение с наставником, который ответит на интересующие вас вопросы.

Текстовая консультация — это общение с наставником в чате. Чат доступен 1 день с момента начала консультации.

Кодревью — это проверка пулреквеста в репозитории вашего проекта. Наставник оставит комментарии к вашему коду.

От выбранного раздела зависит, кто из наставников сможет провести консультацию.

Вы можете отправить заявку на консультацию конкретному наставнику, или её возьмёт первый, освободившийся из живой очереди.
